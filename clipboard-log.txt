function renderHistory(history) {
  const list = document.getElementById('history');
  list.innerHTML = '';
  history.forEach(item => {
    const li = document.createElement('li');
    li.textContent = item;
    list.appendChild(li);
  });

});

  ipcRenderer.send('clear-clipboard-history');
